//创建一个空目录

$ mkdir learngit
$ cd learngit
$ pwd		//显示当前目录

$ git init	//把当前目录变成git可以管理的仓库

$ git add	//把文件添加到仓库
$ git commit	//把文件提交到仓库

$ git status	//查看仓库当前状态

$ git diff	//查看修改内容

$ git log	//查看版本历史记录

$ git reset --hard HEAD^	//回退到上一个版本

$ git reflog	//记录每一次命令

$ git checkout --learngit.txt	//把learngit.txt文件在工作区的修改全部撤销

$ rm		//删除文件（工作区）

$ git rm 	//删除文件（版本库）
$ git commit
